#  Commercial Agreement
#########################################################################
country_event = {
	id = DH_soviet.1
	title = DH_soviet.1.t
	desc = DH_soviet.1.d
	picture = GFX_news_event_002
	fire_only_once = yes

	is_triggered_only = yes

	option = { #accept deal
		name = DH_soviet.1.a
		ai_chance = {
			factor = 1000
			modifier = {
				SOV = { has_government = democratic }
				factor = 0
			}
		}
		GER = {
			58 = {
				add_resource = {
						type = oil
						amount = 12
						state = 58
				}
				add_resource = {
						type = chromium
						amount = 6
						state = 58
				}
			}
		}
		SOV = {
			add_ideas = GER_commercial
		}
	}
	option = { #refuse deal
		ai_chance = { factor = 1 }
		name = DH_soviet.1.b
		GER = { country_event = { hours = 6 id = DH_germany.11 }}
	}
}

#########################################################################
country_event = {
	id = DH_soviet.6
	title = DH_soviet.6.t
	desc = DH_soviet.6.d
	picture = GFX_news_event_002
	fire_only_once = no
	
	trigger = {
		tag = SOV
		SOV = {
			surrender_progress > 0.25
		}
		GER = { controls_state = 219 
		controls_state = 195 
		controls_state = 217 
		controls_state = 229
		controls_state = 653
		if = {
			limit = {
				16 = {
					OR = {
						is_controlled_by = GER
						any_other_country = {
							controls_state = 16
							is_in_faction_with = GER
							GER = { is_faction_leader = yes }
						}
					}
				}
			}
		}		

	}
	
	# We surrender! Let us keep what we still have
	option = { 
		name = DH_soviet.6.a
		country_event = { days = 31 id = DH_germany.25 }
		GER = {
			country_event = { id = GER.16 }
		}
		SOV = { country_event = { id = DH_soviet.3 }
		}
	}
	# Fight to the bitter end!
	option = { 
		name = DH_soviet.6.b
		country_event = { days = 31 id = GER.26 }
	}	
}
